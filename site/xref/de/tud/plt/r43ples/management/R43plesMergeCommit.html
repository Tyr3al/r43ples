1   package de.tud.plt.r43ples.management;
2   
3   import java.util.regex.Matcher;
4   import java.util.regex.Pattern;
5   
6   import org.apache.log4j.Logger;
7   
8   import de.tud.plt.r43ples.exception.InternalErrorException;
9   
10  public class R43plesMergeCommit extends R43plesCommit {
11  	
12  	/** default logger for this class */
13  	private final static Logger logger = Logger.getLogger(R43plesMergeCommit.class);
14  	
15  	private final int patternModifier = Pattern.DOTALL + Pattern.MULTILINE + Pattern.CASE_INSENSITIVE;
16  	private final Pattern patternMergeQuery = Pattern.compile(
17  			"(?<action>MERGE|REBASE|MERGE FF)\\s*(?<type>AUTO|MANUAL)?\\s*GRAPH\\s*<(?<graph>[^>]*?)>\\s*(SDD\\s*<(?<sdd>[^>]*?)>)?\\s*BRANCH\\s*\"(?<branchNameA>[^\"]*?)\"\\s*INTO\\s*\"(?<branchNameB>[^\"]*?)\"(?<with>\\s*WITH\\s*\\{(?<triples>.*)\\})?",
18  			patternModifier);
19  	public final String triples;
20  	public final String branchNameA;
21  	public final String branchNameB;
22  	public final String sdd;
23  	public final String graphName;
24  	public final String type;
25  	public final String action;
26  	public final boolean with;
27  
28  	public R43plesMergeCommit(R43plesRequest request) throws InternalErrorException {
29  		super(request);
30  		Matcher m = patternMergeQuery.matcher(request.query_sparql);
31  		if (!m.find())
32  			throw new InternalErrorException("Error in query: " + request.query_sparql);
33  		
34  		action = m.group("action");
35  		type = m.group("type");
36  		graphName = m.group("graph");
37  		sdd = m.group("sdd");
38  		branchNameA = m.group("branchNameA").toLowerCase();
39  		branchNameB = m.group("branchNameB").toLowerCase();
40  		with = m.group("with")!=null;
41  		triples = m.group("triples");
42  		
43  		logger.debug("type: " + type);
44  		logger.debug("graph: " + graphName);
45  		logger.debug("sdd: " + sdd);
46  		logger.debug("branchNameA: " + branchNameA);
47  		logger.debug("branchNameB: " + branchNameB);
48  		logger.debug("with: " + with);
49  		logger.debug("triples: " + triples);
50  	}
51  	
52  	
53  	public R43plesMergeCommit(final String graphName,
54  			final String branchNameA,
55  			final String branchNameB,
56  			final String user,
57  			final String message,
58  			final String format
59  			) {
60  		super(new R43plesRequest("", format));
61  		this.graphName = graphName;
62  		this.branchNameA = branchNameA;
63  		this.branchNameB = branchNameB;
64  		this.user = user;
65  		this.message = message;
66  		this.sdd = null;
67  		this.action = null;
68  		this.triples = null;
69  		this.type = null;
70  		this.with = false;
71  	}
