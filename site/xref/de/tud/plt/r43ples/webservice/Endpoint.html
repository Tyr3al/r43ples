90  	 * HTTP POST interface for query and update (e.g. SELECT, INSERT, DELETE).
91  	 * 
92  	 * @param formatHeader
93  	 *            format specified in the HTTP header
94  	 * @param formatQuery
95  	 *            format specified in the HTTP parameters
96  	 * @param sparqlQuery
97  	 *            the SPARQL query
98  	 * @param query_rewriting
99  	 * 			  should query rewriting option be used
100 	 * @return HTTP response
101 	 * @throws InternalErrorException 
102 	 */
103 	@POST
104 	@Produces({ MediaType.TEXT_PLAIN, MediaType.TEXT_HTML, MediaType.APPLICATION_JSON, "application/rdf+xml", "text/turtle", "application/sparql-results+xml"})
105 	@Consumes(MediaType.APPLICATION_FORM_URLENCODED)
106 	public final Response sparqlPOST(
107 			@FormParam("format") final String formatQuery, 
108 			@FormParam("query") @DefaultValue("") final String sparqlQuery,
109 			@FormParam("query_rewriting") @DefaultValue("") final String query_rewriting) throws InternalErrorException {
110 		try {
111 			String format = getFormat(formatQuery);
112 			logger.info("SPARQL POST query (format: "+format+", query: "+sparqlQuery +")");
113 			return sparql(format, sparqlQuery, query_rewriting);
114 		} catch (Exception e) {
115 			return Response.serverError().status(Response.Status.NOT_ACCEPTABLE).build();
116 		}
117 	}

119 	private String getFormat(final String formatQuery) throws Exception {
120 		if (formatQuery == null){
121 			List<Variant> reqVariants = Variant.mediaTypes(MediaType.TEXT_PLAIN_TYPE, MediaType.TEXT_HTML_TYPE, 
122 					MediaType.APPLICATION_JSON_TYPE, TEXT_TURTLE_TYPE, APPLICATION_RDF_XML_TYPE, APPLICATION_SPARQL_RESULTS_XML_TYPE).build();
123 			Variant bestVariant = request.selectVariant(reqVariants);
124 	        if (bestVariant == null) {
125 	        	throw new Exception("Requested datatype not available");
126 	        }
127         	MediaType reqMediaType = bestVariant.getMediaType();
128         	return reqMediaType.toString();
129 		}
130 		else {
131 			return formatQuery;
132 		}
133 	}

136 	 * HTTP POST interface for query and update (e.g. SELECT, INSERT, DELETE).
137 	 * Direct method (http://www.w3.org/TR/2013/REC-sparql11-protocol-20130321/#query-via-post-direct)
138 	 * 
139 	 * @param formatHeader
140 	 *            format specified in the HTTP header
141 	 * @param sparqlQuery
142 	 *            the SPARQL query specified in the HTTP POST body
143 	 * @return HTTP response
144 	 * @throws InternalErrorException 
145 	 */
146 	@POST
147 	@Produces({ MediaType.TEXT_PLAIN, MediaType.TEXT_HTML, MediaType.APPLICATION_JSON, "application/rdf+xml", "text/turtle", "application/sparql-results+xml" })
148 	@Consumes("application/sparql-query")
149 	public final Response sparqlPOSTdirectly(
150 			final String sparqlQuery) throws InternalErrorException {
151 		List<Variant> reqVariants = Variant.mediaTypes(MediaType.TEXT_PLAIN_TYPE, MediaType.TEXT_HTML_TYPE, 
152 				MediaType.APPLICATION_JSON_TYPE, TEXT_TURTLE_TYPE, APPLICATION_RDF_XML_TYPE, APPLICATION_SPARQL_RESULTS_XML_TYPE).build();
153 		Variant bestVariant = request.selectVariant(reqVariants);
154         if (bestVariant == null) {
155             return Response.serverError().status(Response.Status.NOT_ACCEPTABLE).build();
156         }
157     	MediaType reqMediaType = bestVariant.getMediaType();
158     	String format = reqMediaType.toString();
159 		logger.info("SPARQL POST query directly (format: "+format+", query: "+sparqlQuery +")");
160 		return sparql(reqMediaType.toString(), sparqlQuery);
161 	}

164 	 * HTTP GET interface for query and update (e.g. SELECT, INSERT, DELETE).
165 	 * Provides HTML form if no query is specified and HTML is requested
166 	 * Provides Service Description if no query is specified and RDF
167 	 * representation is requested
168 	 * 
169 	 * @param formatHeader
170 	 *            format specified in the HTTP header
171 	 * @param formatQuery
172 	 *            format specified in the HTTP parameters
173 	 * @param sparqlQuery
174 	 *            the SPARQL query
175 	 * @param query_rewriting
176 	 * 			  should query rewriting option be used
177 	 * @return HTTP response
178 	 * @throws InternalErrorException 
179 	 */
180 	@GET
181 	@Produces({ MediaType.TEXT_PLAIN, MediaType.TEXT_HTML, MediaType.APPLICATION_JSON, "application/rdf+xml", "text/turtle", "application/sparql-results+xml" })
182 	public final Response sparqlGET(
183 			@QueryParam("format") final String formatQuery, 
184 			@QueryParam("query") @DefaultValue("") final String sparqlQuery,
185 			@QueryParam("query_rewriting") @DefaultValue("") final String query_rewriting) throws InternalErrorException {
186 		String format;
187 		try {
188 			format = getFormat(formatQuery);
189 		} catch (Exception e) {
190 			return Response.serverError().status(Response.Status.NOT_ACCEPTABLE).build();
191 		}		
192 		
193 		String sparqlQueryDecoded;
194 		try {
195 			sparqlQueryDecoded = URLDecoder.decode(sparqlQuery, "UTF-8");
196 		} catch (UnsupportedEncodingException e) {
197 			e.printStackTrace();
198 			sparqlQueryDecoded = sparqlQuery;
199 		}
200 		logger.info("SPARQL GET query (format: "+format+", query: "+sparqlQueryDecoded +")");
201 		return sparql(format, sparqlQueryDecoded, query_rewriting);
202 	}

207 	 * Interface for query and update (e.g. SELECT, INSERT, DELETE).
208 	 * Provides HTML form if no query is specified and HTML is requested
209 	 * Provides Service Description if no query is specified and RDF
210 	 * representation is requested
211 	 * 
212 	 * @param format
213 	 *            mime type for response format
214 	 * @param sparqlQuery
215 	 *            decoded SPARQL query
216 	 * @param query_rewriting
217 	 * 			  should query rewriting option be used
218 	 * @return the response
219 	 * @throws InternalErrorException 
220 	 */
221 	public final Response sparql(final String format, final String sparqlQuery, final boolean query_rewriting) throws InternalErrorException {
222 		if ("".equals(sparqlQuery)) {
223 			if (format.contains(MediaType.TEXT_HTML)) {
224 				return getHTMLResponse();
225 			} else {
226 				return getServiceDescriptionResponse(format);
227 			}
228 		} else {
229 			return getSparqlResponse(format, sparqlQuery, query_rewriting);
230 		}
231 	}

234 	 * 
235 	 * @param format
236 	 *            mime type for response format
237 	 * @param sparqlQuery
238 	 *            decoded SPARQL query
239 	 * @param query_rewriting
240 	 * 			  string determining if query rewriting option be used
241 	 * @return
242 	 * @throws InternalErrorException
243 	 */
244 	private final Response sparql(final String format, final String sparqlQuery, final String query_rewriting) throws InternalErrorException {
245 		String option = query_rewriting.toLowerCase();
246 		if (option.equals("on") || option.equals("true") || option.equals("new"))
247 			return sparql(format, sparqlQuery, true);
248 		else
249 			return sparql(format, sparqlQuery, false);
250 	}

278 	 * Get HTML response for standard sparql request form.
279 	 * Using mustache templates. 
280 	 * 
281 	 * @return HTML response for SPARQL form
282 	 */
283 	private Response getHTMLResponse() {
284 		logger.info("SPARQL form requested");
285 		MustacheFactory mf = new DefaultMustacheFactory();
286 	    Mustache mustache = mf.compile("templates/endpoint.mustache");
287 	    StringWriter sw = new StringWriter();
288 		Map<String, Object> htmlMap = new HashMap<String, Object>();
289 	    htmlMap.put("graphList", RevisionManagement.getRevisedGraphsList());
290 	    htmlMap.put("endpoint_active", true);
291 	    mustache.execute(sw, htmlMap);		
292 		String content = sw.toString();
293 		return Response.ok().entity(content).type(MediaType.TEXT_HTML).build();
294 	}

311 	 * @param result result from the attached triplestore

314 		MustacheFactory mf = new DefaultMustacheFactory();
315 		Mustache mustache = mf.compile("templates/result.mustache");
316 		StringWriter sw = new StringWriter();
317 		Map<String, Object> htmlMap = new HashMap<String, Object>();

322 		mustache.execute(sw, htmlMap);		
323 		return sw.toString();

327 	/**
328 	 * @param format
329 	 * 			requested mime type 
330 	 * @param sparqlQuery
331 	 * 			string containing the SPARQL query
332 	 * @param query_rewriting
333 	 * 			  should query rewriting option be used
334 	 * @return HTTP response of evaluating the sparql query 
335 	 * @throws InternalErrorException
336 	 */
337 	private Response getSparqlResponse(String format, String sparqlQuery, final boolean query_rewriting) throws InternalErrorException {
338 		logger.debug(String.format("SPARQL request (format=%s, query_rewriting=%s) -> %n %s", format, query_rewriting, sparqlQuery));
339 		
340         
341 		R43plesRequest request = new R43plesRequest(sparqlQuery, format);
342 
343 		String result;
344 		if (request.isSelectAskConstructQuery()) {
345 			result = Interface.sparqlSelectConstructAsk(request, query_rewriting);
346 		}
347 		else if (request.isUpdateQuery()) {
348 			Interface.sparqlUpdate(new R43plesCommit(request));
349 			result = "Query executed";
350 		}
351 		else if (request.isCreateGraphQuery()) {
352 			String graphName = Interface.sparqlCreateGraph(sparqlQuery);
353 			result = "Graph <"+graphName+"> successfully created";
354 		}
355 		else if (request.isDropGraphQuery()) {
356 			Interface.sparqlDropGraph(sparqlQuery);
357 			result = "Graph successfully dropped";
358 		}
359 		else if (request.isBranchOrTagQuery()) {
360 			Interface.sparqlTagOrBranch(new R43plesCommit(request));
361 			result = "Tagging or branching successful";
362 		}
363 		else if (request.isMergeQuery()) {
364 			return getMergeResponse(new R43plesMergeCommit(request));
365 		}
366 		else
367 			throw new QueryErrorException("No R43ples query detected");
368 
369 		ResponseBuilder responseBuilder = Response.ok();
370 		if (format.equals("text/html")){
371 			if (query_rewriting) {
372 				responseBuilder.entity(getHTMLResult(result, sparqlQuery, SparqlRewriter.rewriteQuery(sparqlQuery)));
373 			}
374 			else {
375 				responseBuilder.entity(getHTMLResult(result, sparqlQuery));
376 			}
377 		} else {
378 			responseBuilder.entity(result);
379 		}
380 		responseBuilder.type(format);
381 		responseBuilder.header("r43ples-revisiongraph", RevisionManagement.getResponseHeaderFromQuery(sparqlQuery));		
382 		return responseBuilder.build();
383 	}

388 	 * Provides the SPARQL Endpoint description of the original sparql endpoint
389 	 * with the additional R43ples feature (sd:feature) and replaced URIs.
390 	 * 
391 	 * @param format
392 	 *            serialisation format of the service description
393 	 * @return Extended Service Description
394 	 */
395 	private Response getServiceDescriptionResponse(final String format) {
396 		logger.info("Service Description requested");
397 		String triples =String.format("@prefix rdf:	<http://www.w3.org/1999/02/22-rdf-syntax-ns#> . %n"
398 				+ "@prefix ns3:	<http://www.w3.org/ns/formats/> .%n"
399 				+ "@prefix sd:	<http://www.w3.org/ns/sparql-service-description#> .%n"
400 				+ "<%1$s>	rdf:type	sd:Service ;%n"
401 				+ "	sd:endpoint	<%1$s> ;%n"
402 				+ "	sd:feature	sd:r43ples ;"
403 				+ "	sd:resultFormat	ns3:SPARQL_Results_JSON ,%n"
404 				+ "		ns3:SPARQL_Results_XML ,%n"
405 				+ "		ns3:Turtle ,%n"
406 				+ "		ns3:N-Triples ,%n"
407 				+ "		ns3:N3 ,%n"
408 				+ "		ns3:RDF_XML ,%n"
409 				+ "		ns3:SPARQL_Results_CSV ,%n"
410 				+ "		ns3:RDFa ;%n"
411 				+ "	sd:supportedLanguage	sd:SPARQL10Query, sd:SPARQL11Query, sd:SPARQL11Query, sd:SPARQL11Update, sd:R43plesQuery  ;%n"
412 				+ "	sd:url	<%1$s> .%n", uriInfo.getAbsolutePath()) ;
413 		Model model = JenaModelManagement.readStringToJenaModel(triples, "TURTLE");
414 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
415 		
416 		if (format.toLowerCase().contains("xml") )
417 			model.write(baos, "RDF/XML");
418 		else if (format.toLowerCase().contains("turtle") )
419 			model.write(baos, "Turtle");
420 		else if (format.toLowerCase().contains("json") )
421 			model.write(baos, "RDF/JSON");
422 		else {
423 			try {
424 				model.write(baos, format);
425 			}
426 			catch (NoWriterForLangException e) {
427 				model.write(baos, "Turtle");
428 			}
429 		}
430 		return Response.ok().entity(baos.toString()).build();
431 	}

435 	 * Creates a merge between the specified branches.
436 	 * 
437 	 * Using command: MERGE GRAPH \<graphURI\> BRANCH "branchNameA" INTO "branchNameB"
438 	 * 
439 	 * @param sparqlQuery the SPARQL query
440 	 * @param format the result format
441 	 * @throws InternalErrorException 
442 	 */
443 	private Response getMergeResponse(final R43plesMergeCommit commit) throws InternalErrorException {
444 		ResponseBuilder responseBuilder = Response.created(URI.create(""));
445 		logger.info("Merge query detected");
446 		
447 		MergeResult mresult = Interface.sparqlMerge(commit);
448 		RevisionGraph graph = new RevisionGraph(mresult.graph);
449 		
450 		if (mresult.hasConflict) {
451 			responseBuilder = Response.status(Response.Status.CONFLICT);
452 			responseBuilder.entity(mresult.conflictModel);
453 		}
454 		String graphNameHeader;
455 		try {
456 			graphNameHeader = URLEncoder.encode(mresult.graph, "UTF-8");
457 		} catch (UnsupportedEncodingException e) {
458 			e.printStackTrace();
459 			graphNameHeader = mresult.graph;
460 		}
461 		
462 		// Return the revision number which were used (convert tag or branch identifier to revision number)
463 		responseBuilder.header(graphNameHeader + "-revision-number-of-branch-A", graph.getRevisionNumber(mresult.branchA));
464 		responseBuilder.header(graphNameHeader + "-revision-number-of-branch-B", graph.getRevisionNumber(mresult.branchB));			
465 		
466 		responseBuilder.header("r43ples-revisiongraph", RevisionManagement.getResponseHeaderFromQuery(commit.query_sparql));	
467 		
468 		return responseBuilder.build();	
469 	}
